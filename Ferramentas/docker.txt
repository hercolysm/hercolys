////////// make.php \\\\\\\\\\\
<?php

$configs = [
	"/var/www/html/projeto/" => "/home/user/projeto/",
];

$str_mount = null;

foreach ($configs as $target => $source) {
	if (!empty($source)) {
		$str_mount.= "--mount type=bind,source={$source},target={$target} ";
	}
}

$str = 'docker run -d --name=container_name '.$str_mount.' image_name';
print $str."\n";

system($str);
