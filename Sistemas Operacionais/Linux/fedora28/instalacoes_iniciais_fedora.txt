Instalações iniciais (Fedora)

# yum check-update // busca atualizacoes

# yum update // atualiza o sistema


# Instalação do Apache
yum install httpd // instala servidor apache
service httpd start // inicia o servico
chkconfig httpd on // configura na inicializacao do sistema

# Instalacao do Mysql
yum install mysql-server // instala mysql
mysql_install_db //
(mysql --initialize) // 
service mysqld start // inicia o servico
chkconfig mysqld on // configura na inicializacao do sistema
mysqladmin -u root password "nova_senha" // define a senha para root
mysqladmin -u root -p password "nova_senha" // alteracao de senha

# Instalação do PHP
yum install php mysql-php // instala o php
yum install php-pdo // instala o pdo
yum install php-pdo_mysql // instala o pdo mysql

# Desativar SELINUX
vim /etc/sysconfig/selinux
SELINUX=disabled

# Instalar o terminator
yum install terminator

# Instalar area de trabalho do gnome
dnf install @mate-desktop
(reiniciar e selecionar o desktop na hora do login)

# instalar o google chrome
cat << EOF > /etc/yum.repos.d/google-chrome.repo
[google-chrome]
name=google-chrome - x86_64
baseurl=http://dl.google.com/linux/chrome/rpm/stable/x86_64
enabled=1
gpgcheck=1
gpgkey=https://dl.google.com/linux/linux_signing_key.pub
EOF
dnf install google-chrome-stable


# instalar sublime
(Install the GPG key)
sudo rpm -v --import https://download.sublimetext.com/sublimehq-rpm-pub.gpg
sudo dnf config-manager --add-repo https://download.sublimetext.com/rpm/stable/x86_64/sublime-text.repo
sudo dnf install sublime-text
